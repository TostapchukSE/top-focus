**Add confirmTask function:**

```javascript
async function confirmTask(text) {
  const trimmed = text.trim();
  if (!trimmed) return; // don't save empty tasks

  currentTask = {
    text: trimmed,
    createdAt: new Date().toISOString()
  };

  await store.set('currentTask', currentTask);
  await store.save();
  renderCompactMode();
  await collapseWidget();
}
```

```javascript
function initTaskControls() {
  const taskInput = document.querySelector('.task-input');
  const btnDone = document.querySelector('.btn-done');
  const btnClear = document.querySelector('.btn-clear');

  // Enter key confirms task (TASK-02)
  taskInput.addEventListener('keydown', (e) => {
    if (e.key === 'Enter') {
      e.preventDefault();
      confirmTask(taskInput.value);
    }
    // Escape cancels and collapses (TASK-04)
    if (e.key === 'Escape') {
      e.preventDefault();
      // Restore original text (discard edits)
      taskInput.value = currentTask ? currentTask.text : '';
      collapseWidget();
    }
  });

  // Also listen for Escape on the document when expanded (in case input is not focused)
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && isExpanded && !isAnimating) {
      e.preventDefault();
      const ti = document.querySelector('.task-input');
      ti.value = currentTask ? currentTask.text : '';
      collapseWidget();
    }
  });

  // Done button (TASK-05 — completion, wired in Task 2 of this plan)
  // Clear button (TASK-08)
  btnClear.addEventListener('click', (e) => {
    e.stopPropagation();
    clearTask();
  });
}
```

```javascript
async function completeTask() {
  if (!currentTask || isAnimating) return;
  isAnimating = true;

  const widget = document.querySelector('.widget');

  // If expanded, collapse content first
  if (isExpanded) {
    widget.classList.add('collapsing');
    widget.classList.remove('expanded');
    const appWindow = window.__TAURI__.window.getCurrentWindow();
    const LogicalSize = window.__TAURI__.dpi
      ? window.__TAURI__.dpi.LogicalSize
      : window.__TAURI__.window.LogicalSize;
    await new Promise(resolve => setTimeout(resolve, TRANSITION_MS));
    widget.classList.remove('collapsing');
    await appWindow.setSize(new LogicalSize(COMPACT_WINDOW.width, COMPACT_WINDOW.height));
    isExpanded = false;
  }

  // Add to history before clearing
  await addToHistory(currentTask);

  // Show completion animation: green bg + checkmark (1.5s)
  const compactContent = document.querySelector('.compact-content');
  const originalHTML = compactContent.innerHTML;

  // Replace compact content with checkmark
  compactContent.innerHTML = '<span class="completion-checkmark">Done &#10003;</span>';
  widget.classList.add('completing');

  // After 1.5s, clear and return to empty state
  setTimeout(async () => {
    widget.classList.remove('completing');
    compactContent.innerHTML = originalHTML;

    // Clear the task
    currentTask = null;
    await store.set('currentTask', null);
    await store.save();
    renderCompactMode();
    isAnimating = false;
  }, 1500);
}
```

**Add history management functions (TASK-07):**
```javascript
const MAX_HISTORY = 20;

async function addToHistory(task) {
  if (!store || !task) return;
  const history = (await store.get('taskHistory')) || [];

  // Add completed task to front of history
  history.unshift({
    text: task.text,
    completedAt: new Date().toISOString()
  });

  // Trim to max 20
  if (history.length > MAX_HISTORY) {
    history.length = MAX_HISTORY;
  }

  await store.set('taskHistory', history);
  await store.save();
}

async function loadHistory() {
  if (!store) return [];
  return (await store.get('taskHistory')) || [];
}

function renderHistory(historyItems) {
  const historySection = document.querySelector('.history-section');
  const historyList = document.querySelector('.history-list');

  if (!historyItems || historyItems.length === 0) {
    historySection.classList.remove('has-items');
    historyList.innerHTML = '';
    return;
  }

  historySection.classList.add('has-items');
  historyList.innerHTML = '';

  historyItems.forEach((item) => {
    const pill = document.createElement('button');
    pill.className = 'history-pill';
    pill.textContent = item.text;
    pill.title = item.text; // tooltip for truncated pills
    pill.addEventListener('click', (e) => {
      e.stopPropagation();
      const taskInput = document.querySelector('.task-input');
      taskInput.value = item.text;
      taskInput.focus();
    });
    historyList.appendChild(pill);
  });
}
```
