```javascript
async function expandWidget() {
  if (isExpanded || isAnimating) return;
  isAnimating = true;

  const widget = document.querySelector('.widget');
  const appWindow = window.__TAURI__.window.getCurrentWindow();
  const LogicalSize = window.__TAURI__.dpi
    ? window.__TAURI__.dpi.LogicalSize
    : window.__TAURI__.window.LogicalSize;

  // Step 1: Resize window FIRST (expand: resize then content)
  await appWindow.setSize(new LogicalSize(EXPANDED_WINDOW.width, EXPANDED_WINDOW.height));

  // Step 2: Update display content before showing expanded view
  const expandedTaskDisplay = document.querySelector('.expanded-task-display');
  const taskInput = document.querySelector('.task-input');
  if (currentTask) {
    expandedTaskDisplay.textContent = currentTask.text;
    taskInput.value = currentTask.text;
  } else {
    expandedTaskDisplay.textContent = '';
    taskInput.value = '';
  }

  // Step 3: Add expanding class (triggers CSS opacity/transform transition)
  widget.classList.add('expanding');

  // Step 4: Force reflow then add expanded class to trigger transition
  void widget.offsetHeight;
  widget.classList.remove('expanding');
  widget.classList.add('expanded');

  isExpanded = true;

  // Focus the input field after transition completes
  setTimeout(() => {
    taskInput.focus();
    isAnimating = false;
  }, TRANSITION_MS);
}
```

```javascript
async function collapseWidget() {
  if (!isExpanded || isAnimating) return;
  isAnimating = true;

  const widget = document.querySelector('.widget');
  const appWindow = window.__TAURI__.window.getCurrentWindow();
  const LogicalSize = window.__TAURI__.dpi
    ? window.__TAURI__.dpi.LogicalSize
    : window.__TAURI__.window.LogicalSize;

  // Step 1: Fade out content FIRST (collapse: content then resize)
  widget.classList.add('collapsing');
  widget.classList.remove('expanded');

  // Step 2: After CSS transition completes, shrink window
  setTimeout(async () => {
    widget.classList.remove('collapsing');
    await appWindow.setSize(new LogicalSize(COMPACT_WINDOW.width, COMPACT_WINDOW.height));
    isExpanded = false;
    isAnimating = false;
  }, TRANSITION_MS);
}
```

**Add opacity control functions:**
```javascript
async function initOpacity() {
  if (!store) return;
  const prefs = await store.get('preferences');
  const opacity = (prefs && typeof prefs.opacity === 'number') ? prefs.opacity : 0.95;
  applyOpacity(opacity);

  const slider = document.querySelector('.opacity-slider');
  slider.value = Math.round(opacity * 100);

  slider.addEventListener('input', async (e) => {
    const val = parseInt(e.target.value, 10) / 100;
    applyOpacity(val);
    // Persist
    const currentPrefs = (await store.get('preferences')) || {};
    currentPrefs.opacity = val;
    await store.set('preferences', currentPrefs);
    await store.save();
  });
}

function applyOpacity(value) {
  const widget = document.querySelector('.widget');
  widget.style.opacity = Math.max(0.3, Math.min(1.0, value));
}
```
